Experiment 27: Displaying Arrays

Documentation 




    Kindly check 'serverApp2.js' for this experiment which has a link at the bottom of this page    
    In this experiment, Wrote a server app which listens to port 3000 to a particular 
    url configured in the code and responds accordingly.
    So the step you need to follow to run this code are:
    - Open the browser
    - type in the url: http://localhost:3000/actors.html 
    OR
    - type in the url: http://localhost:3000/actress.html 

    The result you get depends on the different strings you type in which can
    be found in 'serverApp2.js' which in this experiment are 'actors' and 'actress'

Code Snippet


var express = require('express')   //imports
var app = express() //instance of expresss
var actors = ['Harry',
'Ron',
'Malfoy'];
var actress = ['Hermoine', 'Luna', 'Parvati', 'Padma'];
app.get("/actors", function (req, res) {
    res.json(actors);
});
app.get("/actress", function (req, res) {
    res.json(actress);
});
app.listen(3000);
