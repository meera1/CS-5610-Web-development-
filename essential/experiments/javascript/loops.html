In this experiment, I used java script to display the 
factorial of a number by using for loop and if loops

    block of code to be executed if the condition is true
} else {

}

for (statement 1; statement 2; statement 3) {
    code block to be executed
}

          function factorial1(txt)
        {
            var n = parseInt(txt);
            if (n < 0)
            {
                alert("Factroial not possible of a negative number");
            }
            else if (n == 0 || n == 1) {
                var n = 1;
                document.getElementById('demo').innerHTML = n;
            }
            else
            {
                var i = 1;
                var fact = 1;
                for (i = 1; i <= n; i++)
                {
                    fact *= i;
                }
                document.getElementById('demo').innerHTML = fact;
            }
        }
