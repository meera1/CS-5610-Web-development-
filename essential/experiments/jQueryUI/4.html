    I have stored all the paragraphs matching with a particular class in my local storage
    after converting them into JSON objects.
    This can be used when you want cookies like functionalities in your website, 
    Suppose you access a website but don't have any network coverage, if the user data was 
    stored in your local storage, having no network connection would not be any issue.

Code Snippet

        <script>
                     $(function () {
            $("#save").click(save);
        });
        function save()
        {
            var allP = $("p.p1");
            //console.log(allP);
            var paragraphs = [];
            allP.each(function (index, item) {
                //console.log(index);
                //console.log(item);
                var p1 = $(item);
                var p2 = p1.text();
                var pos = p1.position();
                //console.log(p2);
                //console.log(pos);
                var pObj = {
                    text: p2,
                    x: pos.left,
                    y: pos.top
                };
                paragraphs.push(pObj); // we store all the paragraphs in the array with their text and position in json format
            });
            console.log(paragraphs);
            pStr = JSON.stringify(paragraphs);
            console.log(pStr);
            localStorage.setItem("page", pStr);
        }
        
        </script>
